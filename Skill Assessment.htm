        function toggleCategory(header) {

            if (content.classList.contains('active')) {

                content.classList.add('active');

                icon.textContent = '−';

            }

        }

            if (e.target.type === 'radio') {

                allOptions.forEach(option => {

                    parent.classList.remove('selected');

                const selectedParent = e.target.closest('.rating-option');

        function generateResults() {

            // Calculate scores for each category

            const categories = {

                'Basic Academic Skills': ['academic_fundamentals', 'classical_texts', 'pathophysiology'],

                'Clinical Skills': ['history_taking', 'physical_exam', 'pulse_diagnosis', 'constitutional_assessment'],

                'Diagnostic & Investigation Skills': ['lab_interpretation', 'imaging_interpretation', 'ecg_interpretation', 'differential_diagnosis'],

                'Treatment & Prescription Skills': ['treatment_planning', 'single_herbs', 'formulations', 'prescription_writing'],

                'Diet & Lifestyle Planning': ['diet_planning', 'lifestyle_counseling', 'seasonal_regimen'],

                'Panchakarma & Surgical Skills': ['panchakarma_knowledge', 'panchakarma_practical', 'surgical_skills', 'parasurgical'],

                'Modern Medicine Knowledge': ['modern_medicine', 'emergency_medicine', 'drug_interactions', 'referral_skills'],

                'Analytical & Reasoning Skills': ['clinical_reasoning', 'critical_thinking', 'evidence_based'],

                'Professional Attitude & Behavior': ['empathy', 'professional_ethics', 'teamwork', 'self_reflection', 'cultural_sensitivity']

            };

 

            let totalScore = 0;

            let totalItems = 0;

            const categoryScores = {};

 

            // Calculate category scores

            Object.keys(categories).forEach(category => {

                const skills = categories[category];

                let categoryTotal = 0;

                let categoryItems = 0;

 

                skills.forEach(skill => {

                    if (assessmentData[skill]) {

                        categoryTotal += parseInt(assessmentData[skill]);

                        categoryItems++;

                    }

                });

 

                if (categoryItems > 0) {

                    const categoryAverage = (categoryTotal / categoryItems);

                    categoryScores[category] = {

                        score: categoryAverage,

                        percentage: (categoryAverage / 5) * 100

                    };

                    totalScore += categoryTotal;

                    totalItems += categoryItems;

                }

            });

 

            const overallPercentage = totalItems > 0 ? ((totalScore / totalItems) / 5) * 100 : 0;

 

            // Display results

            displayResults(overallPercentage, categoryScores);

        }

 

        function displayResults(overallPercentage, categoryScores) {

            // Display overall score

            document.getElementById('overallScore').textContent = Math.round(overallPercentage) + '%';

           

            let level = '';

            if (overallPercentage >= 90) {

                level = 'Excellent - Outstanding Performance';

                levelColor = '#28a745';

            } else if (overallPercentage >= 80) {

                level = 'Good - Above Average Performance';

                levelColor = '#17a2b8';

            } else if (overallPercentage >= 70) {

                level = 'Average - Satisfactory Performance';

                levelColor = '#ffc107';

            } else if (overallPercentage >= 60) {

                level = 'Below Average - Needs Improvement';

                levelColor = '#fd7e14';

            } else {

                level = 'Poor - Significant Improvement Required';

                levelColor = '#dc3545';

            }

 

            const overallLevelElement = document.getElementById('overallLevel');

            overallLevelElement.textContent = level;

            overallLevelElement.style.color = levelColor;

 

            // Display category results

            const categoryResultsHTML = Object.keys(categoryScores).map(category => {

                return `
